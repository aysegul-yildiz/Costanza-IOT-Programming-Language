function>> findDayofWeek(year, month, day){
	monthKeys <- {1,4,4,0,2,5,0,3,6,1,4,6}
	lastTwoofYear <- year % 100
	oneQuarter <- lastTwoofYear / 4
	m <- month -1
	dayofWeek <- lastTwoofYear + oneQuarter + day + monthKeys[m]
	if(year<1900){
	dayofWeek <- dayofWeek + 2
}
if(year<1800){
	dayofWeek <- dayofWeek + 4
}
if(year>2000 && year<2099){
	dayofWeek <- dayofWeek - 1
}
	dayofWeek <- dayofWeek % 7
	return dayofWeek
	
}
